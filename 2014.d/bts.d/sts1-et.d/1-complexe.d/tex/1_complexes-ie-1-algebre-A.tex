  \subsection*{Énoncé}
  \textbf{Calculer les expressions suivantes avec un résultat sous la forme $a + jb$ avec $a$, $b$ des réels et $j$ tel que $j^{2} = -1$.}

  \begin{enumerate}
  \item $z_{1} + z_{2}$
  \item $z_{0} - z_{3} + z_{2}$
  \item $z_{6}+ z_{1} + z_{4}$
  \item $z_{7} \times z_{2}$
  \item $z_{2} + z_{7} \times z_{2}$
  \item $(z_{2})^{2}$
  \item $(z_{1} + z_{2})^{2}$
  \item $\dfrac{1}{z_{1}}$
  \item $\dfrac{z_1}{z_5}$
  \item $\dfrac{z_9}{z_2}$
  \end{enumerate}
